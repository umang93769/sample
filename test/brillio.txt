describe('ButtonComponentComponent', () => {
  let component: ButtonComponentComponent;

  beforeEach(() => {
    component = new ButtonComponentComponent();
  });

  it('should set showMessage to true when handleClick is called', () => {
    component.handleClick();
    expect(component.showMessage).toBeTruthy();
  });

  it('should set message to "Button clicked!" when handleClick is called', () => {
    component.handleClick();
    expect(component.message).toEqual('Button clicked!');
  });
});
